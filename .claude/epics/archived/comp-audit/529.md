# 529 — Agent Persona Framing (FR-C3)

## Description

Add persona identity paragraphs to all 6 debate agent prompts. Each agent receives a
~60-80 token identity section prepended to its system prompt that establishes a distinct
professional archetype. This improves debate quality by giving each agent a clear voice
and analytical lens, reducing generic responses and increasing the diversity of perspectives
in the ensemble.

## Acceptance Criteria

1. All 6 agent prompts have a persona identity paragraph prepended
2. Persona text is ~60-80 tokens per agent
3. `# VERSION:` comment bumped in each file
4. Existing prompt body and `PROMPT_RULES_APPENDIX` preserved unchanged
5. No structural changes to prompt modules — text additions only
6. All existing agent tests pass

## Technical Details

**Pattern — Prepend persona before existing body:**
```python
TREND_SYSTEM_PROMPT = (
    """## Your Identity: Momentum Trader
You are a seasoned momentum trader who lives and dies by trend-following signals...
"""
    + _EXISTING_PROMPT_BODY
    + PROMPT_RULES_APPENDIX
)
```

**`tests/unit/agents/test_personas.py`:**

```python
import pytest
from options_arena.agents.prompts.trend_agent import TREND_SYSTEM_PROMPT
from options_arena.agents.prompts.volatility import VOL_SYSTEM_PROMPT
from options_arena.agents.prompts.flow_agent import FLOW_SYSTEM_PROMPT
from options_arena.agents.prompts.fundamental_agent import FUNDAMENTAL_SYSTEM_PROMPT
from options_arena.agents.prompts.risk import RISK_SYSTEM_PROMPT
from options_arena.agents.prompts.contrarian_agent import CONTRARIAN_SYSTEM_PROMPT
from options_arena.agents.prompts.common import PROMPT_RULES_APPENDIX


PROMPTS = {
    "trend": (TREND_SYSTEM_PROMPT, "Momentum Trader"),
    "volatility": (VOL_SYSTEM_PROMPT, "Vol Arb Specialist"),
    "flow": (FLOW_SYSTEM_PROMPT, "Institutional Flow Analyst"),
    "fundamental": (FUNDAMENTAL_SYSTEM_PROMPT, "Event-Driven Analyst"),
    "risk": (RISK_SYSTEM_PROMPT, "Portfolio Risk Manager"),
    "contrarian": (CONTRARIAN_SYSTEM_PROMPT, "Devil's Advocate Strategist"),
}


class TestAgentPersonas:
    """Verify persona identity sections in all agent prompts."""

    @pytest.mark.parametrize("agent_name,prompt_data", PROMPTS.items())
    def test_persona_identity_present(
        self, agent_name: str, prompt_data: tuple[str, str]
    ) -> None:
        """Each prompt contains its persona identity section."""
        prompt, persona_name = prompt_data
        assert f"## Your Identity: {persona_name}" in prompt

    @pytest.mark.parametrize("agent_name,prompt_data", PROMPTS.items())
    def test_rules_appendix_present(
        self, agent_name: str, prompt_data: tuple[str, str]
    ) -> None:
        """Each prompt still contains PROMPT_RULES_APPENDIX."""
        prompt, _ = prompt_data
        assert PROMPT_RULES_APPENDIX in prompt

    @pytest.mark.parametrize("agent_name,prompt_data", PROMPTS.items())
    def test_persona_before_rules(
        self, agent_name: str, prompt_data: tuple[str, str]
    ) -> None:
        """Persona section appears before PROMPT_RULES_APPENDIX."""
        prompt, persona_name = prompt_data
        persona_idx = prompt.index(f"## Your Identity: {persona_name}")
        rules_idx = prompt.index(PROMPT_RULES_APPENDIX)
        assert persona_idx < rules_idx

    @pytest.mark.parametrize("agent_name,prompt_data", PROMPTS.items())
    def test_persona_not_too_long(
        self, agent_name: str, prompt_data: tuple[str, str]
    ) -> None:
        """Persona text is approximately 60-80 tokens (< 500 chars as proxy)."""
        prompt, persona_name = prompt_data
        start = prompt.index(f"## Your Identity: {persona_name}")
        # Find end of persona section (next ## or substantial content)
        persona_section = prompt[start:start + 600]
        # Just verify it's not excessively long
        lines = persona_section.split("\n")
        persona_lines = []
        for line in lines[1:]:  # Skip header
            if line.startswith("##") or line.startswith("You are a"):
                if persona_lines:  # Already collected some lines
                    break
            persona_lines.append(line)
        assert len(persona_lines) <= 10  # Reasonable paragraph length

    def test_all_six_agents_covered(self) -> None:
        """All 6 debate agents have personas."""
        assert len(PROMPTS) == 6

    def test_personas_are_unique(self) -> None:
        """Each agent has a distinct persona name."""
        persona_names = [name for _, name in PROMPTS.values()]
        assert len(set(persona_names)) == 6
```
